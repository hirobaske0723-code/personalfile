# Google Calendar GAS Integration Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** GAS Web App 経由で今週のGoogleカレンダーイベントを取得し、weekly-schedule.html に表示する。

**Architecture:** GASでCalendarAppを使って今週イベントをJSONで返すWeb Appを作成。HTMLはlocalStorageにGAS URLを保存し、初回のみユーザーに入力を求める。以降はページロード時に自動取得。

**Tech Stack:** Google Apps Script (CalendarApp), localStorage, fetch API

- [ ] **Step 1: gcal-gas.js を作成**

```javascript
// Google Apps Script — Googleカレンダー今週イベント取得
// 使い方: script.google.com で新規プロジェクト作成 → このコードを貼り付け → デプロイ

function doGet(e) {
  const cal = CalendarApp.getDefaultCalendar();
  const now = new Date();

  // 今週の月曜00:00〜日曜23:59を計算
  const mon = new Date(now);
  mon.setDate(now.getDate() - ((now.getDay() + 6) % 7));
  mon.setHours(0, 0, 0, 0);
  const sun = new Date(mon);
  sun.setDate(mon.getDate() + 6);
  sun.setHours(23, 59, 59, 999);

  const events = cal.getEvents(mon, sun);

  const items = events.map(function(ev) {
    var isAllDay = ev.isAllDayEvent();
    return {
      summary: ev.getTitle(),
      start: isAllDay
        ? { date: Utilities.formatDate(ev.getStartTime(), Session.getScriptTimeZone(), 'yyyy-MM-dd') }
        : { dateTime: ev.getStartTime().toISOString() }
    };
  });

  var output = ContentService
    .createTextOutput(JSON.stringify({ items: items }))
    .setMimeType(ContentService.MimeType.JSON);

  return output;
}
```

- [ ] **Step 3: loadCalendar() と GAS URL 管理関数を更新**

既存の `loadCalendar()` 関数を以下に置き換え:

```javascript
const GCAL_LS_KEY = 'base_gcal_gas_url';

function getGcalUrl() {
  return localStorage.getItem(GCAL_LS_KEY) || '';
}

function saveGcalUrl() {
  const url = document.getElementById('gcal-url-input').value.trim();
  if (!url.startsWith('https://script.google.com')) {
    alert('GAS Web AppのURLを入力してください');
    return;
  }
  localStorage.setItem(GCAL_LS_KEY, url);
  document.getElementById('gcal-setup').style.display = 'none';
  loadCalendar();
}

async function loadCalendar() {
  const el = document.getElementById('sched-list');
  const gasUrl = getGcalUrl();

  if (!gasUrl) {
    document.getElementById('gcal-setup').style.display = 'block';
    el.innerHTML = '<div class="empty-msg">GAS URLを設定するとカレンダーが表示されます</div>';
    renderLocalItems();
    return;
  }

  try {
    const res = await fetch(gasUrl, { redirect: 'follow' });
    const data = await res.json();
    const items = data.items || [];
    el.innerHTML = '';
    if (items.length === 0) {
      el.innerHTML = '<div class="empty-msg">今週のGoogleカレンダーの予定はありません</div>';
    } else {
      items.forEach(ev => {
        const start = ev.start.dateTime || ev.start.date;
        const d = new Date(start);
        const isAllDay = !ev.start.dateTime;
        const dateStr = fmt(d) + (isAllDay ? '' : ` ${String(d.getHours()).padStart(2,'0')}:${String(d.getMinutes()).padStart(2,'0')}`);
        const div = document.createElement('div');
        div.className = 'event-item';
        div.innerHTML = `
          <div class="event-dot"></div>
          <div class="event-date">${dateStr}</div>
          <div class="event-name">${escHtml(ev.summary || '（無題）')}</div>
        `;
        el.appendChild(div);
      });
    }
  } catch {
    el.innerHTML = '<div class="empty-msg">カレンダーの取得に失敗しました — <span style="cursor:pointer;color:rgba(99,179,237,0.7);" onclick="resetGcalUrl()">URL再設定</span></div>';
  }
  renderLocalItems();
}

function resetGcalUrl() {
  localStorage.removeItem(GCAL_LS_KEY);
  loadCalendar();
}
```
